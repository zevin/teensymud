  def leave(e)
    ch = get_object(e.from)
    characters(e.from).each do |x|
      add_event(location, x.id,:show, ch.name + " has left #{e.msg}.") if x.account
    end
    # remove character
    get_object(location).delete_contents(ch.id)
    ch.location = nil
    add_event(id, to_room, :arrive, ch.id)
  end
